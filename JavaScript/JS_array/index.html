Array in JavaScript

Used to store a collection on data or values

        syntax: let/var/const array_name = [];

        Example: 
            let book = ["math", "physics", "bio", "computer"];
                            (or)
            let book = new Array("math", "physics", "bio", "computer");
            console.log(book[2]);

            let book = ["math", true, 1, "%%"];

Operations On Array

Array Length 

        syntax: 
                array_name.length;
        
        Example: 
                let names = ["varun", "john", "rohn", "wing"];
                console.log("Array Length: " + names.length);

Add element to the Array after the last index

        syntax: 
                array_name.push('element');
        
        Example: 
                let names = ["varun", "john", "rohn", "wing"];
                names.push("kiran");

Add element to the Array in first index

        syntax: 
                array_name.unshift('element');
        
        Example: 
                let names = ["varun", "john", "rohn", "wing"];
                names.unshift("kiran");

Remove the Elements in Array from last

        syntax: 
                array_name.pop();
        
        Example: 
                let names = ["varun", "john", "rohn", "wing"];
                names.pop();

Remove the Elements in Array from first

        syntax: 
                array_name.shift();
        
        Example: 
                let names = ["varun", "john", "rohn", "wing"];
                names.shift();

Remove the Elements in Array from middle

        syntax: 
                array_name.splice(start_index, howmany_elements_to_remove);
        
        Example: 
                let names = ["varun", "john", "rohn", "wing"];
                names.splice(2, 2);

To Empty an Array 

        syntax: 
                array_name = [];
                     (or)
                array_name.length = 0;
        
        Example: 
                let names = ["varun", "john", "rohn", "wing"];
                names = [];
                    (or)
                names.length = 0;

Find the Position of a particular element in an Array 

        syntax: 
                array_name.indexOf(element);
        
        Example: 
                let names = ["varun", "john", "rohn", "wing"];
                let leng = names.indexOf("john");

Check if it is an Array or not 

        syntax: 
                Array.isArray('name');
        
        Example: 
                let names = ["varun", "john", "rohn", "wing"];
                let leng = 4;
                console.log(Array.isArray(names));
                console.log(Array.isArray(leng));

split a string to array elements 

        syntax: 
                array_name.split('where to split');
        
        Example: 
                let txt = "this is a paragraph broken into array";
                let ary = txt.split(' ');

join array elements into a string 

        syntax: 
                array_name.join('character to join two array elements');
        
        Example: 
                let arry = ["this", "is", "an", "array", "joined", "like", "a", "string"];
                let txt = arry.join(' ');

join two or more arrays 

        syntax: 
                let new_array = array1.concat(array2); // if more arrays then array1.concat(array2, array3, ...)
        
        Example: 
                let array1 = ["one", "two", "three"];
                let array2 = [1, 2, 3];
                let new_array = array1.concat(array2);

Multi Dimensional Array

multi-dimensional array

        syntax: 
                let array = [[],[]];        
        Example: 
                let multi_array = [
                   ["wings of fire", "300/-"],
                   ["MCA", "400/-"],
                   ["Soul of Happiness", "900/-"]
                ];

                [    0   1
                  0[00, 01],
                  1[10, 11],
                  2[20, 21]
                ]

Loop through Array

loop through array

        syntax: 
                for(initialization start index of array; condition less than array length; increment/decrement) {
                        array[value of the initialized variable];
                }      
        Example: 
                let loop_array = ["hello", "world", "!!", "welcome"];
                let length = loop_array.length;
                for( let i = 0; i < length; i++) {
                        console.log(`loop_array[${i}]: ${loop_array[i]} \n`);
                }

forEach() 

        syntax: 
                array_name.forEach(function_name);
                function function_name(value) {
                        value; // value of each array Element
                }
        Example: 
                loop_array = ["hello", "world", "!!", "welcome"];
                loop_array.forEach(func);
                function func(value) {
                        console.log(value);
                }
