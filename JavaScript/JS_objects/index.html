Object in JavaScript

Object

  1. Properties
  2. Action or Function
        syntax: 
            let object_name = {};

        Example: 
        let object = {
            firstName: "charan",
            lastName: "raj",
            age: 19
        };
        console.log(object);
        console.log(object.firstName); // accessing using dot notation
        console.log(object['lastName']); //accessing using array notation
            
        -> to modify
        object.firstName = "jathin";

Property

add a property to object 

        syntax: 
            let object_name = {};
            object_name.property = value;

        Example:

            age: 19
        };

this is used to create new objects with same properties

                this.property_name = value(approprate parameter)
            }

Object Prototype

the way through which meathod and properties can be added after the object is made (inheriting takes place)

        syntax:

            function Object(parameters) {
                this.property_name = value(approprate parameter);
            }
            let object_name = new Object(na, na, ... parameters);

            Object.prototype.new_property =  value;

        Example: 

            function Pro(first, last) {
                this.firstName = first;
                this.lastName = last;
            }
            Pro.prototype.nationality = "indian";
            Pro.prototype.name = function() {
                return `${this.firstName} ${this.lastName}`;
            }
            let pro1 = new Pro("Charan", "Raj");
            let pro2 = new Pro("kiran", "Kumar");
            
            
            console.log(pro1);
            console.log(pro1.nationality);
            console.log(pro2);
            console.log(pro2.name());
