Functions in JavaScript

Functions 

        syntax: function function_name() {

        } //function definition
        function_name(); //function call

        Example: 
            function func() {
                console.log("hello world");
            }
            func();

Functions with parameters/arguments 

        syntax: function function_name(parameters list) {

        } //function definition
        function_name(argument list); //function call

        Example: 
            function table(num) {
                for(let i = 1; i <= 10; i++) {
                    console.log(`${num} X ${i} = ${num * i} \n`);
                }
            }
            table(3);

            function sum(a, b, c) {
                console.log("sum = " + (a + b + c));
            }
            sum(4, 4, 12);

arguments property in function 

        syntax: function function_name() {
            arguments;
        } //function definition
        function_name(argument list); //function call

        Example: 
            function argu() {
                console.log(arguments.length);
            }
            argu(2);
            argu("hello", 5);
            argu(2, 4, 5, true);

return in function 


        syntax: function function_name() {
            return (value); (or) return value;
        } //function definition
        let variable = function_name(); //function call

        Example: 
           function multi(a, b) {
            return (a*b);
           }
           console.log(multi(3, 10));

           function compare(a, b) {
            if(a > b) {
                return `${a} is greater`;
            }
            else if( a < b) {
                return `${b} is greater`;
            }
            else if( a == b) {
                return `a and b is equal`;
            }
        }
        console.log(compare( 30, 40));

global and local variable

  • global variables are declared outside the function and can be used throught out the program
  • local variables are declared inside the function and can be used only inside the function 
        let a = 30;  //global variable
        function adding() {
            let b = 20; //local variable
            console.log(a + b);
        }
        adding();

Immediately invoked function

        syntax:
        (function(){//code})();

        example:
        (function() {
            console.log("this console is function is immediately invoked")
        })();

Arrow function

        syntax:
        function_name = () => {
            //code
        }

        example:
        sub = (a, b) => {
            console.log(a - b);
        }
        sub(19, 16);
